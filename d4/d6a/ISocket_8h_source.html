346 ISocket::WaitForData(SOCKET *pSocketForReceiving, SOCKET *pSocketForSending,

348 {
349 int_t nSocketsReady = 0;
350
351 fd_set FdSetReceive; FD_ZERO(&FdSetReceive);
352 fd_set FdSetSend; FD_ZERO(&FdSetSend);
353 fd_set FdSetError; FD_ZERO(&FdSetError);
354
355 if (pSocketForReceiving) {
356 FD_SET(*pSocketForReceiving, &FdSetReceive);
357 }
358 if (pSocketForSending) {
359 FD_SET(*pSocketForSending, &FdSetSend);
360 }
361 if (pSocketForExceptions) {
362 FD_SET(*pSocketForExceptions, &FdSetError);
363 }
364
365 TIMEVAL tv;
366 TIMEVAL *ptv = nullptr;
367
368 if (_m_timeout) {
369 tv.tv_sec = _m_timeout / 1000;
370 tv.tv_usec = _m_timeout * 1000 - tv.tv_sec * 1000000;
371 ptv = &tv;
372 } else {
373 ptv = nullptr; // nullptr for blocking operation (never times out)
374 }
375
376 nSocketsReady = select(0, &FdSetReceive, &FdSetSend, &FdSetError, ptv);
377
378 //If the operation timed out, set a more natural error message
379 if (nSocketsReady == 0) {
380 SetLastError(WSAETIMEDOUT);
381 nSocketsReady = SOCKET_ERROR;
382 }
383
384 return nSocketsReady;
385 }
