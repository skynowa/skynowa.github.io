218 for ( ; ; ) {
219 // find from left two first chars '%'
220 std::csize_t startSepPos = sRv.find(sep);
221 xCHECK_DO(startSepPos == std::tstring_t::npos, break);
222
223 std::csize_t stopSepPos = sRv.find(sep, startSepPos + sep.size());
224 xCHECK_DO(stopSepPos == std::tstring_t::npos, break);
225
226 // copy %var% to temp string
227 std::ctstring_t &rawEnvName = String::cut(sRv, startSepPos, stopSepPos + sep.size());
228 xTEST(!rawEnvName.empty());

235 // replace envVar(%var%) by expandedEnvVar
