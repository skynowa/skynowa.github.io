147 int_t iRv = 0;
148
149 struct sigaction action;
150 {
151 // Block other terminal-generated signals while handler runs
152 sigset_t blockMask;
153 {
155 #if xENV_APPLE
156 iRv = sigemptyset(&blockMask);
157 #else
158 iRv = ::sigemptyset(&blockMask);
159 #endif
160 xTEST_DIFF(iRv, - 1);
161
162 for (const auto &it : a_signalNums) {
163 if (it == SIGKILL || it == SIGSTOP) {
164 continue;
165 }
166
168 #if xENV_APPLE
169 iRv = sigaddset(&blockMask, it);
170 #else
171 iRv = ::sigaddset(&blockMask, it);
172 #endif
173 xTEST_DIFF(iRv, - 1);
174 }
175 }
176
177 action.sa_sigaction = a_onInfo;
178 action.sa_mask = blockMask;
179 action.sa_flags = SA_RESTART | SA_SIGINFO;
180 }
181
182 for (const auto &it : a_signalNums) {
183 switch (it) {
184 case SIGKILL:
185 Trace() << Format::str(xT("xLib: Signal {} ({}) cannot be caught or ignored"),
186 xLEX_TO_STR(SIGKILL), SIGKILL);

190 xLEX_TO_STR(SIGSTOP), SIGSTOP);

194 break;
195 }

197 iRv = ::sigaction(it, &action, nullptr);
