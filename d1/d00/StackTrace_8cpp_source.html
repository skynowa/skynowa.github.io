77void_t
78StackTrace::_format(
79 std::vector<std::vec_tstring_t> &a_stack, ///< stack as std::vector
80 std::tstring_t *a_stackStr ///< [out] stack as formatted string
81) const
82{
83 xCHECK_DO(a_stack.empty(), return);
84 xCHECK_DO(a_stackStr == nullptr, return);
85
86 std::tstring_t sRv;
87 std::vector<int_t> maxs(::columnsNum, 0);
88
89 // add header
90 {
91 std::cvec_tstring_t stackHeader
92 {
93 xT("Module"),
94 xT("File"),
95 xT("Line"),
96 xT("Offset"),
97 xT("Function")
98 };
99
100 a_stack.insert(a_stack.cbegin(), stackHeader);
101 }
102
103 // get elements max sizes
104 for (size_t i = 0; i < ::columnsNum; ++ i) {
105 for (const auto &it : a_stack) {
106 std::csize_t current = it.at(i).size();
107 xCHECK_DO(static_cast<int_t>(current) > maxs[i], maxs[i] = static_cast<int_t>(current));
108 }
109 }
110
111 // formatting
112 std::size_t lineNumber {};
113
114 for (auto it = a_stack.cbegin(); it != a_stack.cend(); ++ it) {
115 if (it != a_stack.cbegin()) {
116 ++ lineNumber;
117 }
118
119 std::tstring_t lineNumberStr;
120 {
121 if (lineNumber == 0) {
122 lineNumberStr = xT("# ");
123 } else {
124 lineNumberStr = Format::str(xT("[{}]"), lineNumber);
125 }
126
127 if (a_stack.size() >= 10 && lineNumber < 10) {
128 lineNumberStr.insert(1, xT(" "));
129 }
130 }
131
132 std::tstringstream_t stackLine;
133 stackLine
134 << std::setw(0) << std::right << lineNumberStr << xT(" ")
135 << std::setw(maxs[0]) << std::left << it->at(0) << xT(" ")
136 << std::setw(maxs[1]) << std::left << it->at(1) << xT(" ")
137 << std::setw(maxs[2]) << std::left << it->at(2) << xT(" ")
138 << std::setw(maxs[3]) << std::left << it->at(3) << xT(" ")
139 << std::setw(0) << std::left << it->at(4);
140
141
142 if (it + 1 != a_stack.cend()) {
143 stackLine << Const::nl();
144 }
145
146 sRv += stackLine.str();
147 }
148
149 // out
150 a_stackStr->swap(sRv);
151}

154StackTrace::_funcArgsDisable(
