95
96 tchar_t *srcPtr = &buffIn[0];
97 size_t srcSize = a_input.size();
98
99 std::vec_tchar_t buff(_buffSize);
100
101 while (srcSize > 0) {
102 tchar_t *dstPtr = &buff[0];
103 std::size_t dstSize = buff.size();
104
105 std::size_t uiRv = ::iconv(_iconv, &srcPtr, &srcSize, &dstPtr, &dstSize);
106 if (uiRv == (size_t)-1) {
107 if (errno == E2BIG) {
108 // ignore this error
109 Cout() << xT("E2BIG - Argument list too long");
110 }
111 else if (_isSkipErrors) {
112 // skip character
113 ++ srcPtr;
114 -- srcSize;
115 } else {
116 _reportError();

118 return false;
119 }
120 }
121
122 sRV.append(&buff[0], buff.size() - dstSize);
123 }
124
125 sRV.swap(*a_output);
126
