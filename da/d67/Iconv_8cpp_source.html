76bool_t
77Iconv::convert(
78 std::ctstring_t &a_input, ///< input string
79 std::tstring_t *a_output ///< [out] result
80) const
81{
82 xTEST_PTR(_iconv);
83 xTEST_PTR(a_output);
84
85 if (_isSkipEncoding) {
86 *a_output = a_input;
87 return true;
88 }
89
90 a_output->clear();
91
92 std::tstring_t sRV;
93
94 std::vec_tchar_t buffIn(a_input.cbegin(), a_input.cend());
95
96 tchar_t *srcPtr = &buffIn[0];
97 size_t srcSize = a_input.size();
98
99 std::vec_tchar_t buff(_buffSize);
100
101 while (srcSize > 0) {
102 tchar_t *dstPtr = &buff[0];
103 std::size_t dstSize = buff.size();
104
105 std::size_t uiRv = ::iconv(_iconv, &srcPtr, &srcSize, &dstPtr, &dstSize);
106 if (uiRv == (size_t)-1) {
107 if (errno == E2BIG) {
108 // ignore this error
109 Cout() << xT("E2BIG - Argument list too long");
110 }
111 else if (_isSkipErrors) {
112 // skip character
113 ++ srcPtr;
114 -- srcSize;
115 } else {
116 _reportError();
117
118 return false;
119 }
120 }
121
122 sRV.append(&buff[0], buff.size() - dstSize);
123 }
124
125 sRV.swap(*a_output);
126
127 return true;
128}
