174HttpClient::HttpCode
175HttpClient::httpCode(
176 cOptionOut &a_optionOut
177) const
178{
179 static const std::map<std::pair<int, int>, HttpCode> httpCodes
180 {
181 {{100, 199}, HttpCode::Info },
182 {{200, 299}, HttpCode::Success },
183 {{300, 399}, HttpCode::Redirection},
184 {{400, 499}, HttpCode::ClientError},
185 {{500, 599}, HttpCode::ServerError}
186 };
187
188 for (const auto &[it_range, it_code] : httpCodes) {
189 const auto &[it_low, it_high] = it_range;
190
191 if ( Algos::isInBounds(a_optionOut.responseCode, it_low, it_high) ) {
192 return it_code;
193 }
194 }
195
196 return HttpCode::Unknown;
197}

446bool_t
447HttpClient::_request(
448 cRequest a_type,
449 OptionIn &a_optionIn,
450 OptionOut *out_optionOut
451)
452{
453 xTEST_DIFF((int)a_type, (int)Request::Unknown);
454 xTEST_PTR(out_optionOut);
455
456 out_optionOut->headers.clear();
457 out_optionOut->body.clear();
458
459 setOption(CURLOPT_PROTOCOLS, CURLPROTO_HTTP | CURLPROTO_HTTPS);
460
461 switch (a_type) {
462 case Request::Get:
463 {
470 a_optionIn.url += xT("?") + a_optionIn.request;
471
472 if (!a_optionIn.isCacheControl) {
473 std::ctstring_t sessId = std::to_string( DateTime().current().toMsec() );
474
475 a_optionIn.url += "&SessId=" + escape(sessId);
476 }
477
478 setOption(CURLOPT_POST, 0L);
479 setOption(CURLOPT_NOBODY, 0L);
480 }
481 break;
482 case Request::Head:
483 {
488 setOption(CURLOPT_NOBODY, 1L);
489 }
490 break;
491 case Request::Post:
492 {
498 setOption(CURLOPT_POST, 1L);
499 setOption(CURLOPT_POSTFIELDS, a_optionIn.request.data());
500 setOption(CURLOPT_POSTFIELDSIZE, a_optionIn.request.size());
501 }
502 break;
503 case Request::Put:
504 {
509 // TODO: Request::Put
510 setOption(CURLOPT_UPLOAD, 1L);
511 }
512 break;
513 case Request::Delete:
514 {
519 setOption(CURLOPT_CUSTOMREQUEST, xT("DELETE"));
520 }
521 break;
522 case Request::Connect:
523 {
528 // TODO: Request::Connect
529 }
530 break;
531 case Request::Options:
532 {
537 // TODO: Request::Options
538 }
539 break;
540 case Request::Trace:
541 {
546 // TODO: Request::Trace
547 }
548 break;
549 case Request::Unknown:
550 xTEST(false);
551 return false;
552 break;
553 }
554
555 std::tstring_t buffRead;
556 std::tstring_t buffHeaderOut;
557 std::tstring_t buffOptionOut;
558 setOptionsDefault(&a_optionIn, buffRead, &buffHeaderOut, &buffOptionOut);
559
560 perform();
561
562 getInfos(out_optionOut);
563
564 // [out]
565 String::split(buffHeaderOut, Const::crNl(), xT(": "), &out_optionOut->headers);
566 out_optionOut->body = buffOptionOut;
567
568 return true;
569}
