145 #elif (xOS_WIN_VER == xOS_WIN_XP)

147 #elif (xOS_WIN_VER == xOS_WIN_S03)
148 osVersion = xT("Server 2003 SP1, Windows XP SP2");
149 #elif (xOS_WIN_VER == xOS_WIN_VISTA)
150 osVersion = xT("Vista, Windows Server 2008");
151 #elif (xOS_WIN_VER == xOS_WIN_7)
152 osVersion = xT("7, Windows Server 2008 R2");
153 #elif (xOS_WIN_VER == xOS_WIN_WIN8)
154 osVersion = xT("8");
155 #elif (xOS_WIN_VER == xOS_WIN_WIN8_1)
156 osVersion = xT("8.1");
157 #elif (xOS_WIN_VER == xOS_WIN_WIN10)
158 osVersion = xT("10");
159 #endif
160 #elif (xOS_ANDROID || xOS_LINUX)
161 cint_t versionMajor = (xOS_LINUX_VER & 0x00FF0000) >> 16;
162 cint_t versionMinor = (xOS_LINUX_VER & 0x0000FF00) >> 8;
163 cint_t versionPatch = (xOS_LINUX_VER & 0x000000FF) >> 0;
164
165 osVersion = Format::str(xT("{}.{}.{}"), versionMajor, versionMinor, versionPatch);
166 xTEST_EQ(LINUX_VERSION_CODE, xLINUX_KERNEL_VER(versionMajor, versionMinor, versionPatch));
167 #elif xOS_FREEBSD
168 osVersion = String::cast(xOS_FREEBSD_VER);
169 #elif xOS_MACOSX
170 // TODO: [skynowa] BuildInfo::os() - OS version
171 osVersion = Const::strUnknown();
172 #endif
173 }
174
175 sRv = Format::str(xT("{} {}"), osName, osVersion);
176 sRv = String::trimRightChars(sRv, Const::space());
177
178 return sRv;
179}
