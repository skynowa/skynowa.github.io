141DnsClient::nameInfo(
142 ISocket::cAddressFamily a_family, ///< address family
143 std::ctstring_t &a_hostAddr, ///< IP address
144 cushort_t a_hostPort, ///< port
145 std::tstring_t *out_name, ///< [out] host name
146 std::tstring_t *out_port ///< [out] service name
147)
148{

152 xTEST_NA(out_name);
153 xTEST_NA(out_port);
154
155 sockaddr_in socketAddr {};
156 socketAddr.sin_family = static_cast<sa_family_t>(a_family);
157 socketAddr.sin_addr.s_addr = ::inet_addr( xT2A(a_hostAddr).c_str() );
158 socketAddr.sin_port = htons(a_hostPort);
159
160 tchar_t hostName[NI_MAXHOST + 1] = {0};
161 tchar_t hostPort[NI_MAXSERV + 1] = {0};
162
163 int_t iRv = xNAME_INFO_GET(reinterpret_cast<sockaddr *>(&socketAddr), sizeof(socketAddr),
164 &hostName[0], NI_MAXHOST, &hostPort[0], NI_MAXSERV, NI_NUMERICSERV);
165#if xENV_WIN
166 xTEST_EQ(iRv, 0);
167#elif xENV_UNIX
168 xTEST_EQ_MSG(iRv, 0, ::gai_strerror(iRv));
169#endif
170
171 // [out]
172 if (out_name != nullptr) {
173 *out_name = hostName;
174 }
175
176 if (out_port != nullptr) {
177 xTEST_EQ(a_hostPort, (ushort_t)std::atoi(hostPort));
178 *out_port = hostPort;
179 }
180}

369DnsClient::isOnLan(
370 cuint_t a_ip
371)
372{
373 cuint_t localIp = INADDR_ANY; // IP of local interfaCe (network order)
374 cuint_t netMask = INADDR_NONE; // net mask for IP (network order)
375
376 return (((ntohl(a_ip) ^ ntohl(localIp)) & ntohl(netMask)) == 0UL);
