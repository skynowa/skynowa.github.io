165bool_t
166Archive::fileUncompress(
167 std::ctstring_t &a_sourceFilePath,
168 std::ctstring_t &a_destDirPath,
169 cbool_t a_isRemoveSourceFile,
170 cbool_t a_isAutoDetectType
171)
172{
173 if ( !a_isAutoDetectType ) {
174 xTEST_DIFF((int_t)_type, (int_t)Type::Unknown);
175 }
176 xTEST(!a_sourceFilePath.empty());
177 xTEST(!a_destDirPath.empty());
178 xTEST_NA(a_isRemoveSourceFile);
179
180 Dir(a_destDirPath).pathCreate();
181
182 const auto type = a_isAutoDetectType ? _detectType(a_sourceFilePath) : _type;
183
184 std::tstring_t binPath;
185 std::vec_tstring_t params;
186 {
187 switch (type) {
188 case Type::Zip:
189 binPath = ::unzipPath;
190 params = {a_sourceFilePath, "-d", a_destDirPath};
191 break;
192 case Type::zip7:
193 binPath = ::zip7Path;
194 params = {"x", a_sourceFilePath, "-o", a_destDirPath};
195 break;
196 case Type::Rar:
197 binPath = ::unrarPath;
198 params = {"x", a_sourceFilePath, a_destDirPath};
199 break;
200 case Type::Gz:
201 xUNUSED(a_isRemoveSourceFile);
202
203 binPath = ::gunzipPath;
204 params = {a_sourceFilePath};
205 break;
206 case Type::TarBz2:
207 binPath = ::tarPath;
208 params = {"xvjf", a_sourceFilePath, "-C", a_destDirPath};
209 break;
210 case Type::Unknown:
211 default:
212 xTEST(false);
213 return false;
214 break;
215 }
216 }
217
218 Process::execute(binPath, params, {}, xTIMEOUT_INFINITE, nullptr, nullptr);
219
220 // remove zip file
221 if (a_isRemoveSourceFile) {
222 if (type == Type::Gz) {
223 // n/a
224 } else {
225 File(a_sourceFilePath).remove();
226 }
227 }
228
229 // fix unzip bug - chmod for dest dir
230 if (type == Type::Zip &&
231 !Dir(a_destDirPath).isExists())
232 {
233 std::ctstring_t binPathFixed = ::chmodPath;
234 std::cvec_tstring_t paramsFixed = {xT("-R"), xT("0777"), a_destDirPath};
235
236 Process::execute(binPathFixed, paramsFixed, {}, xTIMEOUT_INFINITE, nullptr, nullptr);
237 }
238
239 return true;
240}
