103/* static */
104std::string
105String::castA(
106 std::cwstring_t &a_str,
107 const std::locale &a_locale
108)
109{
110 xTEST_NA(a_str);
111 xTEST_NA(a_locale);
112 xSTD_VERIFY(std::has_facet<std::ctype<wchar_t> >(a_locale));
113 xSTD_VERIFY(std::has_facet<std::ctype<char> >(a_locale));
114
115 xCHECK_RET(a_str.empty(), std::string());
116 xCHECK_RET(!std::has_facet<std::ctype<wchar_t> >(a_locale), std::string());
117 xCHECK_RET(!std::has_facet<std::ctype<char> >(a_locale), std::string());
118
119 using state_type_t = std::wstring::traits_type::state_type;
120 using codecvt_t = std::codecvt<wchar_t, char, state_type_t>;
121
122 std::string asRv(a_str.size(), std::wstring::value_type());
123
124 const codecvt_t &codec = std::use_facet<codecvt_t>( a_locale );
125 state_type_t state {};
126
127 cptr_cwchar_t itBegin = &a_str.at(0);
128 cptr_cwchar_t itEnd = &a_str.at(0) + a_str.size();
129 const wchar_t *itNext = nullptr;
130
131 char *itToBegin = &asRv.at(0);
132 char *itToEnd = &asRv.at(0) + asRv.size();
133 char *itToNext = nullptr;
134
135 codec.out(state, itBegin, itEnd, itNext, itToBegin, itToEnd, itToNext);
136
137 return asRv;
138}

468void_t
469String::split(
470 std::ctstring_t &a_str,
471 std::ctstring_t &a_sep,
472 std::vec_tstring_t *a_rv
473)
474{
475 xTEST_NA(a_str);
476 xTEST_NA(a_sep);
477 xTEST_NA(a_rv);
478
479 xCHECK_DO(a_rv != nullptr, a_rv->clear());
480
481 xCHECK_DO(a_str.empty(), return);
482 xCHECK_DO(a_sep.empty(), return);
483 xTEST_PTR(a_rv);
484
485 std::vec_tstring_t vsRv;
486 std::size_t posPrev = 0U; // start of string
487
488 for ( ; ; ) {
489 std::csize_t pos = a_str.find(a_sep, posPrev);
490 xCHECK_DO(pos == std::tstring_t::npos, break);
491
492 vsRv.push_back(a_str.substr(posPrev, pos - posPrev));
493
494 posPrev = pos + a_sep.size();
495 }
496
497 vsRv.push_back( a_str.substr(posPrev, a_str.size() - posPrev) );
498
499 // out
500 a_rv->swap(vsRv);
501}

651)
652{
653 xTEST_NA(a_bytes);
654
655 std::tstring_t sRv = Const::strUnknown();
656
657 culonglong_t eb = xEB(1);
658 culonglong_t pb = xPB(1);
659 culonglong_t tb = xTB(1);
660 culonglong_t gb = xGB(1);
661 culonglong_t mb = xMB(1);
662 culonglong_t kb = xKB(1);
663 culonglong_t byte = xBYTES(1);
664
665
666 if ( a_bytes / eb > 0ULL) {

668 }
669 else if (a_bytes / pb > 0ULL) {
670 sRv = FormatC::str(xT("%.2f PB"), static_cast<double>(a_bytes) / static_cast<double>(pb));
671 }
672 else if (a_bytes / tb > 0ULL) {
673 sRv = FormatC::str(xT("%.2f TB"), static_cast<double>(a_bytes) / static_cast<double>(tb));
674 }
675 else if (a_bytes / gb > 0ULL) {
676 sRv = FormatC::str(xT("%.2f GB"), static_cast<double>(a_bytes) / static_cast<double>(gb));
677 }
678 else if (a_bytes / mb > 0ULL) {
679 sRv = FormatC::str(xT("%.2f MB"), static_cast<double>(a_bytes) / static_cast<double>(mb));
680 }
681 else if (a_bytes / kb > 0ULL) {
682 sRv = FormatC::str(xT("%.2f KB"), static_cast<double>(a_bytes) / static_cast<double>(kb));
683 }
684 else if (a_bytes / byte > 0ULL) {

686 }
