24template<typename T>
25T
26String::castTo(
27 std::ctstring_t &a_value
28)
29{
30 T rv {};
31
32 try {
33 // bool_t
34 if (std::is_same<T, bool_t>::value) {
35 rv = static_cast<T>( std::stoul(a_value) );
36 }
37
38 // tchar_t
39 else if (std::is_same<T, tchar_t>::value ||
40 std::is_same<T, uchar_t>::value)
41 {
42 if ( a_value.size() > 0) {
43 rv = static_cast<T>(a_value[0]);
44 }
45 }
46
47 // short_t
48 else if (std::is_same<T, short_t>::value) {
49 rv = static_cast<T>( std::stoi(a_value) );
50 }
51 else if (std::is_same<T, ushort_t>::value) {
52 rv = static_cast<T>( std::stoul(a_value) );
53 }
54
55 // int_t
56 else if (std::is_same<T, int_t>::value) {
57 rv = static_cast<T>( std::stoi(a_value) );
58 }
59 else if (std::is_same<T, uint_t>::value) {
60 rv = static_cast<T>( std::stoul(a_value) );
61 }
62
63 // long_t
64 else if (std::is_same<T, long_t>::value) {
65 rv = static_cast<T>( std::stol(a_value) );
66 }
67 else if (std::is_same<T, ulong_t>::value) {
68 rv = static_cast<T>( std::stoul(a_value) );
69 }
70
71 // longlong_t
72 else if (std::is_same<T, longlong_t>::value) {
73 rv = static_cast<T>( std::stoll(a_value) );
74 }
75 else if (std::is_same<T, ulonglong_t>::value) {
76 rv = static_cast<T>( std::stoull(a_value) );
77 }
78
79 // float_t
80 else if (std::is_same<T, float_t>::value) {
81 rv = static_cast<T>( std::stof(a_value) );
82 }
83 else if (std::is_same<T, double_t>::value) {
84 rv = static_cast<T>( std::stod(a_value) );
85 }
86 else if (std::is_same<T, longdouble_t>::value) {
87 rv = static_cast<T>( std::stold(a_value) );
88 }
89
90 // enum
91 else if (std::is_enum<T>::value) {
92 rv = static_cast<T>( std::stoll(a_value) );
93 }
94
95 // other
96 else {
97 static_assert(true, "Unknown type");
98 xTEST_FAIL_MSG(xT("Unknown type"));
99 }
100 }
101 catch (const std::exception &a_exp) {
102 xTEST_FAIL_MSG( std::tstring_t(a_exp.what()) );
103
104 rv = {};
105 }
106 catch (...) {
107 xTEST_FAIL_MSG(xT("Unknown exception"));
108
109 rv = {};
110 }
111
112 return rv;
113}

247
248 xCHECK_DO(a_str.empty(), return);

251 xCHECK_DO(a_map == nullptr, return);
252
253 std::vec_tstring_t lines;

255
256 for (const auto &it_line : lines) {

260 xCHECK_DO(pos == std::tstring_t::npos, continue);

262 std::ctstring_t &key = it_line.substr(0, pos);
263 std::ctstring_t &value = it_line.substr(pos + a_sepKeyValue.size(), std::tstring_t::npos);
264
