146std::tstring_t
147ConsoleUI::_boxLine(
148 std::ctstring_t &a_text,
149 std::csize_t a_width
150) const
151{
152 std::ctstring_t border = xT("|");
153 ctchar_t space = xT(' ');
154 std::ctstring_t paddingLeft = border + space;
155 std::ctstring_t padingRight = xT(" ") + border;
156 std::ctstring_t dot3 = xT(" ...");
157
158 std::tstring_t line = paddingLeft + a_text;
159
160 std::ssize_t delta = static_cast<std::ssize_t>(a_width - line.size());
161 if (delta < 0) {
162 line.resize(a_width - padingRight.size() - dot3.size()); // set padding
163 line += dot3;
164 } else {
165 std::ctstring_t add(static_cast<std::size_t>(delta) - padingRight.size(), space);
166
167 line += add;
168 }
169
170 line += padingRight;
171
172 return line;
173}
