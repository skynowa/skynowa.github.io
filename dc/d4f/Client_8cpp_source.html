253Client::onReadData(
254 void_t *out_buff,
255 std::csize_t a_size,
256 std::csize_t a_items,
257 void_t *out_userData
258)
259{
260 xTEST_PTR(out_buff);
261 xTEST_DIFF(a_size, std::size_t{0});
262 xTEST_DIFF(a_items, std::size_t{0});
263 xTEST_PTR(out_userData);
264
265 auto *data = static_cast<struct ReadData *>(out_userData);
266 xTEST_PTR(data);
267
268 std::ctstring_t buff = data->buff.substr(data->bytes);
269 if ( buff.empty() ) {
270 return 0;
271 }
272
273 std::csize_t buffSizeMax = a_size * a_items;
274 std::size_t buffSize = buff.size();
275 if (buffSizeMax < buffSize) {
276 buffSize = buffSizeMax;
277 }
278
279 std::memcpy(out_buff, buff.data(), buffSize);
280
281 data->bytes += buffSize;
282
283 return buffSize;
284}
