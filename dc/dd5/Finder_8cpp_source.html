189void_t
190Finder::files(
191 std::ctstring_t &a_rootDirPath,
192 std::ctstring_t &a_shellFilter,
193 cbool_t a_isRecursively,
194 std::vec_tstring_t *a_filePaths
195)
196{
197 xTEST(!a_rootDirPath.empty());
198 xTEST(!a_shellFilter.empty());
199 xTEST_NA(a_isRecursively);
200 xTEST_PTR(a_filePaths);
201
202 if (!a_isRecursively) {
203 for (Finder finder(a_rootDirPath, a_shellFilter); finder.moveNext(); ) {
204 xCHECK_DO(finder.isDot(), continue);
205
206 // set filter for files
207 if (static_cast<FileType::types_t>(FileType::Type::Directory) & finder.fileTypes()) {
208 continue;
209 }
210
211 a_filePaths->push_back(finder.entryPath());
212 }
213 } else {
214 // subdirs
215 std::vec_tstring_t dirPaths;
216 dirs(a_rootDirPath, Const::maskAll(), true, &dirPaths);
217
218 // files in root dir and each subdir
219 files(a_rootDirPath, a_shellFilter, false, a_filePaths);
220
221 for (const auto &it : dirPaths) {
222 files(it, a_shellFilter, false, a_filePaths);
223 }
224 }
