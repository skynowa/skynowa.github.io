479void_t
480DateTime::_construct(
481 culonglong_t a_msec
482)
483{
484 xTEST_NA(a_msec);
485
486 // datetime msec member
487 this->_thisMSec = a_msec;
488
489 // datetime members
490 ulonglong_t msec = a_msec;
491
492 // converts milliseconds to datetime members
493 _year = static_cast<int_t>( msec / ((1000ULL * 60 * 60 * 24 * 30) * 12) );
494
495 msec %= ((1000ULL * 60 * 60 * 24 * 30) * 12);
496 _month = static_cast<int_t>( msec / ((1000ULL * 60 * 60 * 24) * 30) );
497
498 msec %= ((1000ULL * 60 * 60 * 24) * 30);
499 _day = static_cast<int_t>( msec / ((1000ULL * 60 * 60) * 24) );
500
501 msec %= ((1000ULL * 60 * 60) * 24);
502 _hour = static_cast<int_t>( msec / ((1000ULL * 60) * 60) );
503
504 msec %= ((1000ULL * 60) * 60);
505 _minute = static_cast<int_t>( msec / ((1000ULL * 60)) );
506
507 msec %= ((1000ULL * 60));
508 _second = static_cast<int_t>( msec / 1000ULL );
509
510 msec %= 1000ULL;
511 _msec = static_cast<int_t>( msec );
512}
