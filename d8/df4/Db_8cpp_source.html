30Db::show(
31 std::ctstring_t &a_sqlLike, ///< simple regular expression (SQL LIKE)
32 std::vec_tstring_t *out_dbNames ///< [out]
33) const
34{
35 xTEST_NA(a_sqlLike);
36 xTEST_PTR(out_dbNames);
37
38 out_dbNames->clear();
39
40 // Connection
41 Connection conn(_options);
42 conn.connect();
43
44 // Query
45 std::tstring_t sql = xT("SHOW DATABASES");
46 if ( !a_sqlLike.empty() ) {
47 sql += Format::str(xT(" LIKE {}"), EscapeQuoted(conn, a_sqlLike).forSqm());
48 }
49
50 // Result
51 {
52 StoreResult result = conn.query(sql).store();
53
54 rows_t rows;
55 result.fetchRows(&rows);
56
57 // [out]
58 for (const auto &it_row : rows) {
59 xTEST_EQ(it_row.size(), static_cast<std::size_t>(1));
60
61 const auto &value = it_row.begin()->second;
62
63 out_dbNames->emplace_back(value);
64 }
65 }
