215template<typename T, HandleType typeT>
216bool_t
217Handle<T, typeT>::isValid() const
218{
219#if xENV_WIN
220 if constexpr (typeT == HandleType::NativeInvalid) {
221 // created but not initialised
222 cbool_t cond1 = (_handle != reinterpret_cast<T>( static_cast<intptr_t>(0xCDCDCDCD) ));
223 // uninitialized locals in VC6 when you compile w/ /GZ
224 cbool_t cond2 = (_handle != reinterpret_cast<T>( static_cast<intptr_t>(0xCCCCCCCC) ));
225 // indicate an uninitialized variable
226 cbool_t cond3 = (_handle != reinterpret_cast<T>( static_cast<intptr_t>(0xBAADF00D) ));
227 // no man's land (normally outside of a process)
228 cbool_t cond4 = (_handle != reinterpret_cast<T>( static_cast<intptr_t>(0xFDFDFDFD) ));
229 // freed memory set by NT's heap manager
230 cbool_t cond5 = (_handle != reinterpret_cast<T>( static_cast<intptr_t>(0xFEEEFEEE) ));
231 // deleted
232 cbool_t cond6 = (_handle != reinterpret_cast<T>( static_cast<intptr_t>(0xDDDDDDDD) ));
233 // compare with error handle value
234 cbool_t cond7 = (_handle != null());
235
236 return (cond1 && cond2 && cond3 && cond4 && cond5 && cond6 && cond7);
237 }
238 else if constexpr (typeT == HandleType::Socket) {
239 return (_handle >= 0);
240 }
241 else {
242 return (_handle != null());
243 }
244#elif xENV_UNIX
245 return (_handle != null());
246#endif
247}
