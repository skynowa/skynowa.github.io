247bool_t
248Console::_isColorized(
249 std::tostream_t &a_stream /* = std::cout */
250) const
251{
252 /**
253 * An index to be used to access a private storage of I/O streams. See
254 * colorize / nocolorize I/O manipulators for details
255 */
256 static const int colorizeIndex = std::ios_base::xalloc();
257 if ( static_cast<bool_t>(a_stream.iword(colorizeIndex)) ) {
258 return true;
259 }
260
261 if ( _isAtty(a_stream) ) {
262 return true;
263 }
264
265 return false;
266}
267//-------------------------------------------------------------------------------------------------
268bool_t
269Console::_isAtty(
270 std::ctostream_t &a_stream
271) const
272{
273 FILE *stdStream = _getStdStream(a_stream);
274 if (stdStream == nullptr) {
275 /**
276 * Unfortunately, fileno() ends with segmentation fault
277 * if invalid file descriptor is passed. So we need to
278 * handle this case gracefully and assume it's not a tty
279 * if standard stream is not detected, and 0 is returned
280 */
281
282 return false;
283 }
284
285 return static_cast<bool_t>( xISATTY(xFILENO(stdStream)) );
286}
