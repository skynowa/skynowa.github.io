349 bool_t
350 Process::_isUrlFull(
351  std::ctstring_t &a_url
352 )
353 {
354  // List of known URL schemes
355  constexpr std::ctstring_view_t schemes[] =
356  {
357  xT("http://"), xT("https://"), xT("ftp://"), xT("ftps://"), xT("sftp://"),
358  xT("mailto:"), xT("www."), xT("file://")
359  };
360 
361  // Check if the URL starts with any of the known schemes
