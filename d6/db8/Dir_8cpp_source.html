93bool_t
94Dir::isExists() const
95{
96 xCHECK_RET(_dirPath.empty(), false);
97
98 FileType type(_dirPath);
99
100 xCHECK_RET(type.get() == static_cast<FileType::types_t>(FileType::Type::Unknown), false);
101
102#if xENV_UNIX
103 if (type.get() == static_cast<FileType::types_t>(FileType::Type::SymbolicLink)) {
104 std::ctstring_t &symLinkPathTo = Path::readSymLink(_dirPath);
105
106 FileType symLinkType(symLinkPathTo);
107 bool_t bRv = symLinkType.isExists(FileType::Type::Directory);
108 xCHECK_RET(!bRv, false);
109
110 return true;
111 }
112#endif
113
114 bool_t bRv = type.isExists(FileType::Type::Directory);
115 xCHECK_RET(!bRv, false);
116
117 return true;
118}

179void_t
180Dir::copy(
181 std::ctstring_t &a_dirPathTo,
182 cbool_t a_failIfExists
183) const
184{
185 xTEST(isExists());
186 xTEST(!a_dirPathTo.empty());
187 xTEST_NA(a_failIfExists);
188
189 // sets attribute "normal"
190 bool_t bRv = Dir(a_dirPathTo).isExists();
191 if (bRv) {
192 FileType(a_dirPathTo).clear();
193 }
194
195 FileType(_dirPath).clear();
196
197 // get lists of files
198 std::vec_tstring_t filePaths;
199
200 filePaths.clear();
201 Finder::files(_dirPath, Const::maskAll(), true, &filePaths);
202
203 // copy
204 xFOR_EACH_R_CONST(std::vec_tstring_t, it, filePaths) {
205 std::tstring_t filePathTo = *it;
206
207 size_t posBegin = filePathTo.find(_dirPath);
208 xTEST_DIFF(posBegin, std::tstring_t::npos);
209
210 filePathTo.replace(posBegin, posBegin + _dirPath.size(), a_dirPathTo);
211
212 Dir( Path(filePathTo).dir() ).pathCreate();
213
214 File(*it).copy(filePathTo, a_failIfExists);
215 }
216
217 // TODO: [skynowa] rollback copy

257Dir::tryRemove(
258 std::csize_t a_attempts,
259 culong_t a_timeoutMsec
260) const
261{
262 xTEST_DIFF(a_attempts, size_t(0U));
263
264 std::csize_t maxAttempts = 100; // MAGIC_NUMBER: maxAttempts
265 std::csize_t realAttempts = (maxAttempts < a_attempts) ? maxAttempts : a_attempts;
266
267 for (size_t i = 0; i < realAttempts; ++ i) {
268 bool_t bRv = isExists();
269 xCHECK_DO(!bRv, break);
270
271 FileType(_dirPath).clear();
272
273 bRv = _tryRemove_impl();
274 xCHECK_DO(bRv, break);
275
276 Thread::currentSleep(a_timeoutMsec);
277 }
278
279 xTEST(!isExists());
